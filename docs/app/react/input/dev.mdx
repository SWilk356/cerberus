## API

<NoteAdmonition description="The Label component must be used within a Field provider." />

```ts showLineNumbers=false
export interface FieldProps {
  disabled?: boolean
  invalid?: boolean
  required?: boolean
  readOnly?: boolean
}

define function Field(props: PropsWithChildren<FieldProps>): ReactNode | null
```

```ts showLineNumbers=false
export interface InputProps extends InputHTMLAttributes<HTMLInputElement> {
  describedBy?: string
  id: string
  size?: 'sm' | 'md' | 'lg'
}

define function Input(props: InputProps): ReactNode | null
```

### Props

The `InputProps` component accepts the following props:

| Name     | Default | Description                                                   |
| -------- | ------- | ------------------------------------------------------------- |
| describedBy     |    | The `id` attribute of the FieldMessage the Input is associated with. |
| id   |    |  An identifier that is shared with the Label `htmlFor` attribute.        |
| size   | `lg`   |  The size of the input field.        |
