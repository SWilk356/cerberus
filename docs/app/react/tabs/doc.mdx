## Usage

<CodePreview preview={<BasicTabsPreview />}>
```tsx title="button.tsx"
import { Tabs, TabList, Tab, TabPanel } from '@cerberus-design/react'

function BasicTabsPreview() {
  return (
    <Tabs active="overview">
      <TabList>
        <Tab id="overview" controls="panel:overview" value="overview">
          Overview
        </Tab>
        <Tab id="features" controls="panel:features" value="features">
          Features
        </Tab>
        <Tab id="pricing" controls="panel:pricing" value="pricing">
          Pricing
        </Tab>
      </TabList>
      <TabPanel id="panel:overview" tab="overview">
        Overview content
      </TabPanel>
      <TabPanel id="panel:features" tab="features">
        Features content
      </TabPanel>
      <TabPanel id="panel:pricing" tab="pricing">
        Pricing content
      </TabPanel>
    </Tabs>
  )
}
```

## API

```ts showLineNumbers=false
export interface TabsProps {
  active?: string
  cache?: boolean
}

define function Tabs(props: TabsProps): ReactNode | null
````

### Props

The `Tabs` component accepts the following props:

| Name     | Default | Description                                                   |
| -------- | ------- | ------------------------------------------------------------- |
| active     |    | The `value` of the tab you want to be active by default. |
| cache | `false`    | Whether to cache the active state of the tabs.    |
