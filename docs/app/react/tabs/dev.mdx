```tsx title="tabs.tsx"
import { Tabs, TabList, Tab, TabPanel } from '@cerberus-design/react'
import { css } from '@cerberus/styled-system/css'

function CustomTabsPreview() {
  const tabData = [
    { id: 'overview', value: 'overview', label: 'Overview' },
    { id: 'features', value: 'features', label: 'Features' },

          rounded: 'full',
        })}
      >

          <Tab
            key={tab.id}
            className={css({
              borderTopLeftRadius: 'full',
              borderTopRightRadius: 'full',
              h: '3.7rem',
              rounded: 'full',
              w: '1/3',
              _before: {
                bgColor: 'initial',
                bottom: 'initial',
                content: '""',
                h: 'initial',
                position: 'initial',
                left: 'initial',
                right: 'initial',
                transitionProperty: 'initial',
                transitionDuration: 'initial',
                transitionTimingFunction: 'initial',
                w: 'initial',
                willChange: 'initial',
                zIndex: 'initial',
              },
              _after: {
                borderTopLeftRadius: 'initial',
                borderTopRightRadius: 'initial',
                bottom: 'initial',
                bgColor: 'initial',
                content: '""',
                left: 'initial',
                position: 'initial',
                right: 'initial',
                h: 'initial',
                transition: 'initial',
                willChange: 'initial',
                w: 'initial',
                zIndex: 'initial',
              },
              _hover: {
                bgColor: 'action.navigation.hover',
                color: 'neutral.surface.initial',
              },
              _selected: {
                bgColor: 'action.bg.initial',
                color: 'action.text.initial',
              },
            })}
            value={tab.value}
          >
            {tab.label}
          </Tab>
        ))}
      </TabList>
      {tabData.map((tab) => (
        <TabPanel key={tab.id} tab={tab.value}>
          {tab.label} content
        </TabPanel>
      ))}
    </Tabs>
  )
}
```

## API

### Tabs

```ts showLineNumbers=false
export interface TabsProps {
  active?: string
  cache?: boolean
  palette?: 'action' | 'secondaryAction'
  id?: string
}

define function Tabs(props: TabsProps): ReactNode
````

#### Props

The `Tabs` component accepts the following props:

| Name     | Default | Description                                                   |
| -------- | ------- | ------------------------------------------------------------- |
| active     |    | The `value` of the tab you want to be active by default. |
| cache | `false`    | Whether to cache the active state of the tabs.    |
| id |    | A unique `id` attribute used when there are multiple Tabs on a single page. |
| palette | `action`    | The color palette to use for the Tabs. |

### TabList

```ts showLineNumbers=false
interface TabListProps extends HTMLAttributes<HTMLDivElement> {
  description: string
}

define function TabList(props: TabListProps): ReactNode
````

#### Props

The `TabList` component accepts the following props:

| Name     | Default | Description                                                   |
| -------- | ------- | ------------------------------------------------------------- |
| description     |    | A description of what the Tabs use case is for. |

### Tab

```ts showLineNumbers=false
export interface TabProps extends ButtonHTMLAttributes<HTMLButtonElement> {
  value: string
}

define function Tab(props: TabProps): ReactNode
````

#### Props

The `Tab` component accepts the following props:

| Name     | Default | Description                                                   |
| -------- | ------- | ------------------------------------------------------------- |
| value     |    | An identifier that is used for both the active state and accessbility attributes. |

### TabPanel

```ts showLineNumbers=false
export interface TabPanelProps extends HTMLAttributes<HTMLDivElement> {
  tab: string
}

define function TabPanel(props: TabPanelProps): ReactNode
````

#### Props

The `TabPanel` component accepts the following props:

| Name     | Default | Description                                                   |
| -------- | ------- | ------------------------------------------------------------- |
| tab     |    | The Tab `value` identifier the panel is associated with. |
