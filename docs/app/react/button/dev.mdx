### With Icon

You can utilize the `ButtonParts` API to use an icon within the button. This will allow you to dynamically show a Spinner when the button has a `pending` state.

<CodePreview preview={<WithIconButtonPreview />}>
```tsx title="button.tsx"
import { ButtonParts } from '@cerberus/react'
import { ArrowDownRight } from '@carbon/icons-react'

function WithIconButton() {
  return (
    <ButtonParts.Root>
      With icon
      <ButtonParts.Icon>
        <ArrowDownRight />
      </ButtonParts.Icon>
    </ButtonParts.Root>
  )
}
```
</CodePreview>

## Customization

You can customize the button by utilizing the `cx` function.

<CodePreview preview={<CustomButtonPreview />}>
```tsx title="button.tsx"
import { Button } from '@cerberus/react'
import { css } from '@cerberus/styled-system/css'

function CustomButton() {
  return (
    <Button
      className={css({
        bgColor: 'yellow.800',
        color: 'black',
        _hover: {
          bgColor: 'yellow.500',
        },
      })}
    >
      Wu-Tang Forever
    </Button>
  )
}
```
</CodePreview>

## API

```ts showLineNumbers=false
export interface ButtonProps extends ButtonHTMLAttributes<HTMLButtonElement> {
  palette?: 'action' | 'secondaryAction' | 'danger'
  usage?: 'filled' | 'outlined' | 'ghost'
  shape?: 'sharp' | 'rounded'
  pending?: boolean
}

define function Button(props: ButtonProps): ReactNode
````

### Props

The `Button` component accepts the following props:

| Name     | Default | Description                            |
| -------- | ------- | -------------------------------------- |
| palette     | `action`   | The color palette of the button. |
| usage | `filled`    | The style treatment of the button.    |
| shape | `sharp`     | The shape of the button.              |
| pending | `false`    | The loading state of the button. When a button has an `Icon` it will replace it with a spinner. When not, is shows a disabled state.     |
