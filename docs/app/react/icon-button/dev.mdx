## Usage

The IconButton API is similar to the [Button](./button), but with the addition of an `ariaLabel` prop.

### Default

<CodePreview preview={<BasicIconButtonPreview />}>
```tsx title="icon-button.tsx"
import { IconButton } from '@cerberus-design/react'
import { ThumbsUp } from '@cerberus/icons'

function BasicIconButtonPreview() {
  return (
    <IconButton ariaLabel="Like something">
      <ThumbsUp />
    </IconButton>
  )
}
```
</CodePreview>

## Customization

You can customize the button by utilizing the `cx` function.

<CodePreview preview={<CustomIconButtonPreview />}>
```tsx title="icon-button.tsx"
import { IconButton } from '@cerberus-design/react'
import { css } from '@cerberus/styled-system/css'
import { TrophyFilled } from '@cerberus/icons'

function CustomIconButtonPreview() {
  return (
    <IconButton
      ariaLabel="Wu-Tang Forever"
      className={css({
        alignItems: 'center',
        display: 'inline-flex',
        bgColor: 'yellow',
        color: 'black',
        h: '3.7rem',
        justifyContent: 'center',
        rounded: 'md',
        w: '3.7rem',
        _hover: {
          bgColor: 'black',
          color: 'yellow',
        },
      })}
    >
      <TrophyFilled size={32} />
    </IconButton>
  )
}
```
</CodePreview>

## API

```ts showLineNumbers=false
export interface IconButtonProps extends ButtonHTMLAttributes<HTMLButtonElement> {
  ariaLabel: string
  palette?: 'action' | 'danger'
  usage?: 'filled' | 'text'
  shape?: 'circle'
}

define function IconButton(props: IconButtonProps): ReactNode | null
````

### Props

The `IconButton` component accepts the following props:

| Name     | Default | Description                                                   |
| -------- | ------- | ------------------------------------------------------------- |
| ariaLabel |    "Icon Button"     | The accessible label for the button. |
| palette     | `action`   | The color palette of the button. |
| usage | `text`    | The style treatment of the button.    |
| shape | `circle`    | The shape of the button.               |
