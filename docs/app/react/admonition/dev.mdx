## Customization

<CodePreview preview={<CustomAdmonition />}>
```tsx title="button.tsx"
import { Admonition, AdmonitionHeading, AdmonitionDescription } from '@cerberus/react'
import { css } from '@cerberus/styled-system/css'

export function CustomAdmonition() {
  const styleProps = {
    palette: 'page',
    usage: 'outlined',
  } as AdmonitionProps

  return (
    <Admonition {...styleProps} className={css({
      bgColor: 'black',
      borderColor: 'yellow',
      color: 'yellow',
    })}>
      <AdmonitionHeading {...styleProps}>When to use</AdmonitionHeading>
      <AdmonitionDescription {...styleProps}>
        Use this component when you want to display a page-level message.{' '}
        <a href="#">Learn more</a>
      </AdmonitionDescription>
    </Admonition>
  )
}
```

| -------- | ------- | -------------------------------------- |
