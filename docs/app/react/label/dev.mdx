## Usage

```tsx title="first-name-input.tsx" {5,6}
import { Field, Label } from '@cerberus/react'

function LabelBasicPreview() {
  return (
    <Field required>
      <Label htmlFor="first_name">First Name</Label>
      <Input id="first_name" placeholder="Type your first name" type="text" />
    </Field>
  )
}
```

## API

<NoteAdmonition description="The Label component must be used within a Field provider." />

```ts showLineNumbers=false
export interface FieldProps {
  disabled?: boolean
  invalid?: boolean
  required?: boolean
  readOnly?: boolean
}

define function Field(props: PropsWithChildren<FieldProps>): ReactNode
```

```ts showLineNumbers=false
export interface LabelProps extends LabelHTMLAttributes<HTMLLabelElement> {
  htmlFor: string
  hidden?: boolean
  size?: 'sm' | 'md'
}

define function Label(props: LabelProps): ReactNode
```

### Props

The `LabelProps` component accepts the following props:

| Name     | Default | Description                                                   |
| -------- | ------- | ------------------------------------------------------------- |
| htmlFor     | null   | The `name` attribute of the Input the Label is associated with. |
| hidden   | false   | Whether the Label content should be visually hidden.                 |
| size     | 'md'    | The size of the Label.         |
