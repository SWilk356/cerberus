## Customization

You can use the Notification components to customize the styles.

<CodePreview preview={<CustomNotification />}>
```tsx title="some-page.tsx"
import { Notification } from '@cerberus-design/react'

function CustomNotification() {
  return (
    <Notification
      className={css({
        bgColor: 'black',
      })}
      id="forever"
      open
    >
      <NotificationHeading
        className={css({
          color: 'yellow',
        })}
      >
        Wu-Tang Clan
      </NotificationHeading>
      <NotificationDescription
        className={css({
          color: 'yellow',
        })}
      >
        It's Method Man, for short Mr. Meth
      </NotificationDescription>
    </Notification>
  )
}
```

## API

```ts showLineNumbers=false
export interface NotificationOptions {
  palette?: 'info' | 'success' | 'warning' | 'danger'
  heading: string
  description?: ReactNode
  onClose?: () => void
}

export interface NotificationProviderValue {
  notify: (options: NotificationOptions) => void
}

define function Notification(props: PropsWithChildren<{}>): ReactNode
````

### Show Method Options

The `notify` method accepts the following options:

| Name     | Default | Description                            |
| -------- | ------- | -------------------------------------- |
| palette | `info` | The palette variant used to theme the modal.           |
| heading | `''` | The heading of the modal. |
| description | `null` | The description of the modal. |
| onClose |  | An optional callback that runs when the user closes the notification. |
