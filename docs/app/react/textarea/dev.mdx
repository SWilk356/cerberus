## Usage

```tsx title="first-name-input.tsx" {7}
import { Field, Label, Textarea, FieldMessage } from '@cerberus/react'

function TextareaBasicPreview() {
  return (
    <Field>
      <Label htmlFor="bio">Bio</Label>
      <Textarea describedBy="help:bio" id="bio" />
      <FieldMessage id="help:bio">
        This will be displayed on your public profile.
      </FieldMessage>
    </Field>
  )
}
```

## Customizing


```tsx title="search-input.tsx"
import { Field, Label, Textarea } from '@cerberus/react'

function TextareaCustomPreview() {
  return (
    <Field required>
      <Label htmlFor="killa_bees">Killa Bees</Label>
      <Textarea
        className={css({
          border: '2px solid',
          borderColor: 'yellow',
          bgColor: 'black',
          color: 'yellow',
        })}
        id="killa_bees"
        placeholder="Protect ya neck!"
      />
    </Field>
  )
}
```

## API

<NoteAdmonition description="The Label component must be used within a Field provider." />

```ts showLineNumbers=false
export interface FieldProps {
  disabled?: boolean
  invalid?: boolean
  required?: boolean
  readOnly?: boolean
}

define function Field(props: PropsWithChildren<FieldProps>): ReactNode
```

```ts showLineNumbers=false
export interface TextareaProps
  extends Omit<TextareaHTMLAttributes<HTMLTextAreaElement>, 'id'> {
  describedBy?: string
  id: string
}

define function Textarea(props: TextareaProps): ReactNode
```

### Props

The `TextareaProps` component accepts the following props:

| Name     | Default | Description                                                   |
| -------- | ------- | ------------------------------------------------------------- |
| describedBy     |    | The `id` attribute of the FieldMessage the Input is associated with. |
| id   |    |  An identifier that is shared with the Label `htmlFor` attribute.        |
